    # –§–æ—Ä–º–∏—Ä—É–µ–º —Å–æ–æ–±—â–µ–Ω–∏—è
    messages = [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": f"{formatted_context}\n\n–í–æ–ø—Ä–æ—Å: {user_message}"}
    ]
    
    # –î–æ–±–∞–≤–ª—è–µ–º –∏—Å—Ç–æ—Ä–∏—é –¥–∏–∞–ª–æ–≥–∞
    if conversation_history:
        for msg in conversation_history[-MAX_CONVERSATION_HISTORY:]:
            messages.append(msg)
    
    try:
        raw_response = None
        
        if AI_PROVIDER == "openai":
            response = client.chat.completions.create(
                model=OPENAI_MODEL,
                messages=messages,
                max_tokens=MAX_TOKENS_RESPONSE,
                temperature=TEMPERATURE
            )
            raw_response = response.choices[0].message.content
        
        elif AI_PROVIDER == "anthropic":
            # –£–±–∏—Ä–∞–µ–º system message (—É Anthropic —Å–≤–æ–π —Ñ–æ—Ä–º–∞—Ç)
            system_msg = messages[0]["content"]
            user_messages = messages[1:]
            
            response = client.messages.create(
                model=ANTHROPIC_MODEL,
                max_tokens=MAX_TOKENS_RESPONSE,
                temperature=TEMPERATURE,
                system=system_msg,
                messages=user_messages
            )
            raw_response = response.content[0].text
        
        elif AI_PROVIDER == "groq":
            response = client.chat.completions.create(
                model=GROQ_MODEL,
                messages=messages,
                max_tokens=MAX_TOKENS_RESPONSE,
                temperature=TEMPERATURE
            )
            raw_response = response.choices[0].message.content
        
        # –ü–∞—Ä—Å–∏–º JSON –∏–∑ –æ—Ç–≤–µ—Ç–∞
        trade_rating = {"my_team_rating": 0.0, "their_team_rating": 0.0}
        response_text = raw_response
        
        # –ò—â–µ–º JSON –±–ª–æ–∫ –≤ –æ—Ç–≤–µ—Ç–µ
        try:
            # –ò—â–µ–º JSON –±–ª–æ–∫ –º–µ–∂–¥—É ```json –∏ ``` –∏–ª–∏ –ø—Ä–æ—Å—Ç–æ { ... }
            import re
            json_match = re.search(r'```json\s*(\{.*?\})\s*```', raw_response, re.DOTALL)
            if not json_match:
                json_match = re.search(r'\{[^{}]*"trade_rating"[^{}]*\{[^{}]*"my_team_rating"[^{}]*"their_team_rating"[^{}]*\}', raw_response, re.DOTALL)
            
            if json_match:
                json_str = json_match.group(1) if json_match.groups() else json_match.group(0)
                parsed = json.loads(json_str)
                if "trade_rating" in parsed:
                    trade_rating = parsed["trade_rating"]
                    # –£–¥–∞–ª—è–µ–º JSON –∏–∑ —Ç–µ–∫—Å—Ç–∞ –æ—Ç–≤–µ—Ç–∞
                    response_text = raw_response.replace(json_match.group(0), "").strip()
        except Exception as e:
            # –ï—Å–ª–∏ –Ω–µ —É–¥–∞–ª–æ—Å—å —Ä–∞—Å–ø–∞—Ä—Å–∏—Ç—å JSON, –æ—Å—Ç–∞–≤–ª—è–µ–º –¥–µ—Ñ–æ–ª—Ç–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è
            print(f"Warning: Could not parse trade_rating from response: {e}")
        
        return {

            "trade_rating": trade_rating
        }
        
    except Exception as e:
        return {
            "response": f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –æ–±—Ä–∞—â–µ–Ω–∏–∏ –∫ AI: {str(e)}",
